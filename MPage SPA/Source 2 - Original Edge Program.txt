        <script>
            var PROG_DOT = '01_meds_dot_date_comb_edge3:Group1';
            var PARAMS_DOT = '^MINE^,12690542.0,180.0';
            var PROG_NICU = '01_bk_NICU_Inf_Edge:Group1';
            var PARAMS_NICU = '^MINE^,12690542.0';
            var PROG_GP = '01_meds_pharm_anvs_edge:Group1';
            var PARAMS_GP = '^MINE^,12690542.0,12690542.0,28766114.0';
            var PROG_SEARCH = '01_meds_pharm_search_edge:Group1';
            var PARAMS_SEARCH = '^MINE^,180876.0,12690542.0,28766114.0';
            if (typeof XMLCclRequest === 'undefined' && window.external && 'XMLCclRequest'in window.external) {
                window.XMLCclRequest = function() {
                    return window.external.XMLCclRequest();
                }
                ;
            }
            var activeTabId = null;

            function activateTab(id) {
                var tabs = document.querySelectorAll('.tab-btn');
                for (var i = 0; i < tabs.length; i++) {
                    tabs[i].classList.remove('active');
                }
                var btn = document.getElementById(id);
                if (btn)
                    btn.classList.add('active');
                activeTabId = id;

            }

            function showMsg(msg) {
                var msgDiv = document.getElementById('frame-msg');
                var iframe = document.getElementById('content-frame');
                if (msgDiv) {
                    msgDiv.style.display = 'flex';
                    msgDiv.textContent = msg;
                }
                if (iframe)
                    iframe.style.display = 'none';
            }

            function showIframe() {
                var msgDiv = document.getElementById('frame-msg');
                var iframe = document.getElementById('content-frame');
                if (msgDiv)
                    msgDiv.style.display = 'none';
                if (iframe)
                    iframe.style.display = 'block';
            }

            function loadTab(program, params, tabId) {
                activateTab(tabId);
                showMsg('Loading\u2026');

                if (typeof XMLCclRequest === 'undefined') {
                    showMsg('ERROR: XMLCclRequest is not available in this environment.');
                    return;
                }

                var xcr = new XMLCclRequest();

                xcr.onreadystatechange = function() {
                    if (xcr.readyState === 4) {
                        if (xcr.status === 200) {
                            var iframe = document.getElementById('content-frame');
                            if (iframe) {
                                iframe.srcdoc = xcr.responseText;
                                showIframe();
                            }
                        } else {
                            var dbg = 'Status:' + xcr.status + ' | prog:' + program + ' | params:' + params + ' | resp:' + xcr.responseText;
                            var bar = document.getElementById('dbg-bar');
                            bar.textContent = dbg;
                            bar.style.display = 'block';
                            showMsg('Error ' + xcr.status + ' - see debug bar above');
                        }
                    }
                }
                ;

                xcr.open('GET', program, true);
                xcr.send(params);
            }

            function loadDOT() {
                loadTab(PROG_DOT, PARAMS_DOT + ',' + Date.now(), 'tab-dot');
            }

            function loadNICU() {
                loadTab(PROG_NICU, PARAMS_NICU + ',' + Date.now(), 'tab-nicu');
            }

            function loadGP() {
                loadTab(PROG_GP, PARAMS_GP + ',' + Date.now(), 'tab-dev1');
            }

            function loadSearch() {
                loadTab(PROG_SEARCH, PARAMS_SEARCH + ',' + Date.now(), 'tab-dev2');
            }

            function loadPlaceholder(tabId, label) {
                activateTab(tabId);
                showMsg(label + ' \u2014 not yet available.');
            }
